-- Generate player ratings based on position with some randomness
INSERT INTO PlayerRating (player_id, scoring_rating, is_active)
SELECT
    player_id,
    CASE
        -- Attackers: 70-95 rating
        WHEN player_position IN ('ATT', 'FW') THEN 70 + (ABS(CHECKSUM(NEWID())) % 26)

        -- Midfielders: 50-80 rating
        WHEN player_position IN ('MID', 'MF') THEN 50 + (ABS(CHECKSUM(NEWID())) % 31)

        -- Defenders: 20-50 rating
        WHEN player_position IN ('DEF', 'DF') THEN 20 + (ABS(CHECKSUM(NEWID())) % 31)

        -- Goalkeepers: 5-15 rating
        WHEN player_position = 'GK' THEN 5 + (ABS(CHECKSUM(NEWID())) % 11)

        -- Default fallback
        ELSE 50
        END as scoring_rating,
    1 as is_active
FROM Player
WHERE NOT EXISTS (
    SELECT 1 FROM PlayerRating pr WHERE pr.player_id = Player.player_id
);
GO

-- Show summary statistics
SELECT
    'Total Player Ratings' as Metric,
    COUNT(*) as Count
FROM PlayerRating
UNION ALL
SELECT
    'Active Players',
    COUNT(*)
FROM PlayerRating
WHERE is_active = 1;

PRINT '';
PRINT 'Ratings by Position:';

SELECT
    p.player_position as Position,
    COUNT(*) as PlayerCount,
    MIN(pr.scoring_rating) as MinRating,
    AVG(pr.scoring_rating) as AvgRating,
    MAX(pr.scoring_rating) as MaxRating
FROM Player p
    INNER JOIN PlayerRating pr ON p.player_id = pr.player_id
WHERE pr.is_active = 1
GROUP BY p.player_position
ORDER BY AVG(pr.scoring_rating) DESC;

PRINT '';
PRINT 'Top 20 Rated Players:';

SELECT TOP 20
    p.player_name as PlayerName,
    t.team_name as Team,
       p.player_position as Position,
    pr.scoring_rating as Rating
FROM Player p
    INNER JOIN PlayerRating pr ON p.player_id = pr.player_id
    INNER JOIN Team t ON p.team_id = t.team_id
WHERE pr.is_active = 1
ORDER BY pr.scoring_rating DESC, p.player_name;
