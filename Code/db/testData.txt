-- Verification
SELECT 
    u.first_name, u.last_name, u.user_role, u.user_status, u.wallet_balance,
    COUNT(DISTINCT bs.slip_id)       AS total_slips,
    COUNT(DISTINCT t.transaction_id) AS total_transactions,
    SUM(CASE WHEN t.transaction_type = 'bet'    THEN t.amount ELSE 0 END) AS total_staked,
    SUM(CASE WHEN t.transaction_type = 'payout' THEN t.amount ELSE 0 END) AS total_paid_out
FROM AppUser u
LEFT JOIN BetSlip bs        ON u.app_user_id = bs.app_user_id
LEFT JOIN SystemTransaction t ON u.app_user_id = t.app_user_id
GROUP BY u.app_user_id, u.first_name, u.last_name, u.user_role, u.user_status, u.wallet_balance
ORDER BY u.app_user_id;

-- Overall revenue check
SELECT
    SUM(CASE WHEN transaction_type = 'bet'    THEN amount ELSE 0 END) AS total_stakes,
    SUM(CASE WHEN transaction_type = 'payout' THEN amount ELSE 0 END) AS total_payouts,
    SUM(CASE WHEN transaction_type = 'bet'    THEN amount ELSE 0 END)
  - SUM(CASE WHEN transaction_type = 'payout' THEN amount ELSE 0 END) AS house_revenue
FROM SystemTransaction;
